#include <unistd.h>
#define PORT 8080
#define MAX_NUMBERS 100 // Assuming a maximum number of numbers to handle

int main(int argc, char const *argv[]) {
    int sock = 0, valread;
    struct sockaddr_in serv_addr;
    int numbers[MAX_NUMBERS];
    int size;

    // Prompt the user to enter the size of the array and its elements
    printf("Enter the number of integers: ");
    scanf("%d", &size);
    printf("Enter %d integers:\n", size);
    for (int i = 0; i < size; i++) {
        scanf("%d", &numbers[i]);
    }

    if ((sock = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
        printf("Socket creation error\n");
        return -1;
    }

    serv_addr.sin_family = AF_INET;
    serv_addr.sin_port = htons(PORT);

    if (inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr) <= 0) {
        printf("\nInvalid address/ Address not supported\n");
        return -1;
    }

    if (connect(sock, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {
        printf("\nConnection Failed\n");
        return -1;
    }

    // Send size of numbers array to server
    send(sock, &size, sizeof(size), 0);

    // Send numbers array to server
    send(sock, numbers, size * sizeof(int), 0);

    // Receive sum from server
    int result;
    valread = read(sock, &result, sizeof(result));

    printf("Sum of the numbers is: %d\n", result);

    return 0;
}

SERVER
#include <netinet/in.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <unistd.h>
#define PORT 8080
#define MAX_NUMBERS 100 // Assuming a maximum number of numbers to handle

// Function to calculate the sum of an array of numbers
int sum(int arr[], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += arr[i];
    }
    return sum;
}

int main(int argc, char const *argv[]) {
    int server_fd, new_socket, valread;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    int opt = 1;
    int numbers[MAX_NUMBERS];
    int size;

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("socket failed");
        exit(EXIT_FAILURE);
    }

    if (setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR | SO_REUSEPORT, &opt, sizeof(opt))) {
        perror("setsockopt");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("bind failed");
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("listen");
        exit(EXIT_FAILURE);
    }

    if ((new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t *)&addrlen)) < 0) {
        perror("accept");
        exit(EXIT_FAILURE);
    }

    // Read size of numbers array from client
    valread = read(new_socket, &size, sizeof(size));

    // Read numbers array from client
    valread = read(new_socket, numbers, size * sizeof(int));

    // Calculate sum of numbers
    int result = sum(numbers, size);

    // Send sum back to client
    send(new_socket, &result, sizeof(result), 0);

    printf("Sum sent to client: %d\n", result);

    return 0;
}
